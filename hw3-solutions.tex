\section*{Problem 8.5.2(a)}
\textbf{Setting.} Let $X_1,\dots,X_n \iid N(\mu,\sigma^2)$ with both $\mu$ and $\sigma^2$ unknown. Construct a confidence interval for $\mu$ with confidence coefficient $\gamma$, using
\[
\bar X_n = \frac{1}{n}\sum_{i=1}^n X_i, \qquad
(S_n')^2 = \frac{1}{n-1}\sum_{i=1}^n (X_i-\bar X_n)^2.
\]

\textbf{Step 1: Pivot.} By the Fisher--Cochran theorem, $\bar X_n$ and $(S_n')^2$ are independent, with
\[
\sqrt{n}\,\frac{\bar X_n-\mu}{\sigma}\sim N(0,1), \qquad \frac{(n-1)(S_n')^2}{\sigma^2}\sim \chi^2_{n-1}.
\]
Dividing the $N(0,1)$ variable by $S_n'/\sigma$ gives the $t$-pivot
\begin{equation}\label{eq:tpivot}
T \;=\; \sqrt{n}\,\frac{\bar X_n-\mu}{S_n'} \;\sim\; t_{n-1},
\end{equation}
whose distribution is free of $\mu$ and $\sigma^2$.

\textbf{Step 2: Invert the pivot.} Let $c = T_{n-1,(1+\gamma)/2}$ be the $(1+\gamma)/2$ quantile of $t_{n-1}$. By symmetry, $\Pr(-c \le T \le c) = \gamma$. Rearranging:
\begin{align*}
-c \le \sqrt{n}\,\frac{\bar X_n-\mu}{S_n'} \le c
&\iff -c\,\frac{S_n'}{\sqrt{n}} \le \bar X_n-\mu \le c\,\frac{S_n'}{\sqrt{n}} \\
&\iff \bar X_n - c\,\frac{S_n'}{\sqrt{n}} \le \mu \le \bar X_n + c\,\frac{S_n'}{\sqrt{n}}.
\end{align*}

\textbf{Step 3: Confidence interval.}
\begin{equation}\label{eq:CImu}
\boxed{\;\; \left(\bar X_n - T_{n-1,(1+\gamma)/2}\,\frac{S_n'}{\sqrt{n}},\; \bar X_n + T_{n-1,(1+\gamma)/2}\,\frac{S_n'}{\sqrt{n}}\right).\;\;}
\end{equation}
Since $t_{n-1}$ is symmetric and unimodal, the equal-tail choice gives the shortest $t$-based CI of coverage $\gamma$.

\medskip
\textit{Example.} With $n=9$, $\bar x=22$, $s'=3$, $\gamma=0.95$: $T_{8,0.975}\approx 2.306$, giving $(19.694, 24.306)$.

\section*{Problem 8.5.4}
\textbf{Setting.} Let $X_1,\dots,X_n \iid N(\mu,\sigma^2)$ with both parameters unknown. Construct a confidence interval for $\sigma^2$ with confidence coefficient $\gamma$.

\textbf{Step 1: Pivot.} The quantity
\[
V \;=\; \frac{(n-1)(S_n')^2}{\sigma^2} \;\sim\; \chi^2_{n-1}
\]
is a pivot because its distribution does not depend on $(\mu,\sigma^2)$.

\textbf{Step 2: Equal-tail cutoffs.} Set $\alpha=1-\gamma$ and choose
\[
a = \chi^2_{n-1,\alpha/2}, \qquad b = \chi^2_{n-1,1-\alpha/2},
\]
so that $\Pr(a \le V \le b) = \gamma$. Equal-tail cutoffs do not minimize CI length (the $\chi^2$ density is asymmetric) but are the standard choice.

\textbf{Step 3: Invert for $\sigma^2$.}
\begin{align*}
a \le \frac{(n-1)(S_n')^2}{\sigma^2} \le b
&\iff \frac{(n-1)(S_n')^2}{b} \le \sigma^2 \le \frac{(n-1)(S_n')^2}{a}.
\end{align*}

\textbf{Confidence interval.}
\begin{equation}\label{eq:CIvar}
\boxed{\;\;\left(\frac{(n-1)(S_n')^2}{\chi^2_{n-1,\,1-\alpha/2}},\;\; \frac{(n-1)(S_n')^2}{\chi^2_{n-1,\,\alpha/2}}\right),\quad \alpha = 1-\gamma.\;\;}
\end{equation}
Taking square roots gives a CI for $\sigma$.

\section*{Problem 8.7.11(a)}
\textbf{Setting.} Let $X_1,\dots,X_n \iid N(\mu,\sigma^2)$ and consider $T_c = c\sum_{i=1}^n (X_i - \bar X_n)^2$ for $c>0$. Find $c$ making $T_c$ unbiased for $\sigma^2$.

\textbf{Solution.} Since $\frac{1}{\sigma^2}\sum(X_i-\bar X_n)^2 \sim \chi^2_{n-1}$ and $\E[\chi^2_{n-1}]=n-1$,
\[
\E\!\left[\sum_{i=1}^n (X_i - \bar X_n)^2\right] = (n-1)\sigma^2,
\]
so $\E[T_c] = c(n-1)\sigma^2$. Setting $\E[T_c]=\sigma^2$:
\[
c(n-1)=1 \;\Longleftrightarrow\; \boxed{\;c=\frac{1}{n-1}.\;}
\]
This recovers the unbiased sample variance $(S_n')^2$. The MLE uses $c=1/n$, which introduces bias $-\sigma^2/n$ but achieves lower variance.

\section*{Problem 8.8.2}
\textbf{Setting.} Let $X\sim\mathrm{Exp}(\theta)$, i.e., $f(x\mid\theta)=\theta e^{-\theta x}$ for $x>0$. Compute $I(\theta)$.

\textbf{Log-likelihood and score.}
\[
\ell(x,\theta) = \log\theta - \theta x, \qquad \dot\ell(x,\theta) = \frac{1}{\theta} - x.
\]
We have $\E_\theta[\dot\ell] = 1/\theta - 1/\theta = 0$.

\textbf{Second derivative.}
\[
\ddot\ell(x,\theta) = -\frac{1}{\theta^2}.
\]

\textbf{Fisher information.}
\begin{equation}\label{eq:IthetaExp}
I(\theta) = -\E_\theta[\ddot\ell] = \boxed{\;\frac{1}{\theta^2}.\;}
\end{equation}
Equivalently, $I(\theta)=\Var_\theta(\dot\ell)=\Var_\theta(X)=1/\theta^2$. For a sample of size $n$: $I_n(\theta)=n/\theta^2$, $\mathrm{CRLB}=\theta^2/n$.

\section*{Problem 8.8.7}
\textbf{Setting.} Let $X_1,\dots,X_n \iid N(\mu,\sigma^2)$ with $\mu$ \emph{known}. Determine whether $\hat\sigma^2 = \frac{1}{n}\sum_{i=1}^n (X_i-\mu)^2$ is efficient.

\textbf{Step 1: Unbiasedness.} Since $(X_i-\mu)/\sigma \iid N(0,1)$, we have $\sum(X_i-\mu)^2/\sigma^2\sim\chi^2_n$, so $\E[\hat\sigma^2]=\sigma^2$.

\textbf{Step 2: Variance.} Using $\Var(\chi^2_n)=2n$:
\begin{equation}\label{eq:varMLE}
\Var(\hat\sigma^2) = \frac{\sigma^4}{n^2}\cdot 2n = \frac{2\sigma^4}{n}.
\end{equation}

\textbf{Step 3: Fisher information for $\theta=\sigma^2$.} For $X\sim N(\mu,\theta)$:
\[
\ell(x,\theta) = -\tfrac{1}{2}\log(2\pi\theta) -\frac{(x-\mu)^2}{2\theta}, \qquad
\ddot\ell = \frac{1}{2\theta^2}-\frac{(x-\mu)^2}{\theta^3}.
\]
Taking $-\E[\cdot]$ with $\E[(X-\mu)^2]=\theta$:
\begin{equation}\label{eq:ISigma2}
I(\theta) = \frac{1}{2\theta^2} = \frac{1}{2\sigma^4}.
\end{equation}

\textbf{Step 4: Compare to CRLB.}
\[
\CRLB = \frac{1}{n\,I(\sigma^2)} = \frac{2\sigma^4}{n} = \Var(\hat\sigma^2).
\]
Since $\hat\sigma^2$ is unbiased and achieves the CRLB,
\[
\boxed{\;\hat\sigma^2 = \tfrac{1}{n}\sum(X_i-\mu)^2 \text{ is an efficient estimator of }\sigma^2.\;}
\]

\section*{Problem 8.8.17}
\textbf{Setting.} Let $T$ be an unbiased estimator of $g(\theta)$. Show $T$ attains the CRLB iff $T - g(\theta) = u(\theta)\sum_{i=1}^n \dot\ell(X_i,\theta)$ a.s.\ for some $u(\theta)$.

\textbf{Setup.} Let $U_n=\sum_{i=1}^n\dot\ell(X_i,\theta)$. Under regularity:
\[
\E_\theta[U_n]=0, \quad \Var_\theta(U_n)=n\,I(\theta), \quad \Cov_\theta(T,U_n)=g'(\theta),
\]
where the covariance identity follows by differentiating $\E_\theta[T]=g(\theta)$ under the integral.

\textbf{Cauchy--Schwarz.} Both $T-g(\theta)$ and $U_n$ are mean-zero, so
\[
[g'(\theta)]^2 = [\Cov_\theta(T,U_n)]^2 \le \Var_\theta(T)\cdot n\,I(\theta),
\]
which is the CRLB inequality.

\textbf{Equality condition.} Equality in Cauchy--Schwarz holds iff $T-g(\theta)$ and $U_n$ are linearly dependent a.s.:
\[
T-g(\theta) = u(\theta)\,U_n(\theta) \quad \text{a.s.}
\]
Taking covariance with $U_n$: $g'(\theta) = u(\theta)\cdot n\,I(\theta)$, so $u(\theta)=g'(\theta)/(n\,I(\theta))$, and
\[
\Var_\theta(T) = u(\theta)^2\cdot n\,I(\theta) = \frac{[g'(\theta)]^2}{n\,I(\theta)} = \CRLB. \qquad\square
\]

\[
\boxed{\; \Var_\theta(T)=\CRLB \iff T - g(\theta) = \frac{g'(\theta)}{n\,I(\theta)}\sum_{i=1}^n \dot\ell(X_i,\theta)\ \text{ a.s.}\;}
\]

\section*{Problem 9.1.1(a)}
\textbf{Setting.} Let $X_1,\dots,X_n \iid N(\mu,1)$ and reject $H_0:\mu=0$ iff $\bar X_n > c$.

Under either hypothesis, $\bar X_n\sim N(\mu,1/n)$, so $\sqrt{n}(\bar X_n-\mu)\sim N(0,1)$.

\textbf{Type I error:}
\[
\alpha(c) = \Pr_{\mu=0}(\bar X_n>c) = 1-\Phi(\sqrt{n}\,c).
\]

\textbf{Type II error:}
\[
\beta(c) = \Pr_{\mu=1}(\bar X_n\le c) = \Phi\!\left(\sqrt{n}(c-1)\right).
\]

\[
\boxed{\;\alpha(c) = 1-\Phi(\sqrt{n}\,c), \qquad \beta(c) = \Phi\!\left(\sqrt{n}(c-1)\right).\;}
\]

As $c\uparrow$: $\alpha\downarrow$ and $\beta\uparrow$. For a test of size $\alpha_0$, set $c=z_{1-\alpha_0}/\sqrt{n}$.
