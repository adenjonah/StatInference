\section*{9.5 \#2}

$n = 9$, $\bar{X}_n = 22$, $\sum (X_i - \bar{X}_n)^2 = 72$. Then
\[
    \sigma'^{2} = \tfrac{72}{8} = 9, \qquad
    \sigma'/\sqrt{n} = 1, \qquad
    U = \tfrac{\bar{X}_n - 20}{\sigma'/\sqrt{n}} = 2 \;\sim\; t_8 \text{ under } H_0.
\]

\textbf{(a)} $H_0\!: \mu \le 20$ vs.\ $H_1\!: \mu > 20$.
Reject if $U > t_{0.95,8} = 1.860$. Since $2 > 1.860$, \textbf{reject $H_0$}.

\bigskip

\textbf{(b)} $H_0\!: \mu = 20$ vs.\ $H_1\!: \mu \neq 20$.
Reject if $|U| > t_{0.975,8} = 2.306$. Since $2 < 2.306$, \textbf{fail to reject $H_0$}.

\bigskip

\textbf{(c)} $95\%$ CI for $\mu$:
\[
    \bar{X}_n \pm t_{0.975,8}\cdot \tfrac{\sigma'}{\sqrt{n}}
    \;=\; 22 \pm 2.306
    \;=\; (19.694,\; 24.306).
\]

\section*{9.5 \#4}

$n = 8$, $\sum X_i = -11.2$, $\sum X_i^2 = 43.7$. Test $H_0\!: \mu = 0$ vs.\ $H_1\!: \mu \neq 0$ at $\alpha = 0.10$.
\begin{align*}
    \bar{X}_n &= -1.4, \\
    \sum (X_i - \bar{X}_n)^2 &= 43.7 - 8(1.96) = 28.02, \\
    \sigma' &= \sqrt{28.02/7} \approx 2.001, \qquad
    \sigma'/\sqrt{8} \approx 0.7074, \\
    U &= \tfrac{-1.4}{0.7074} \approx -1.979 \;\sim\; t_7.
\end{align*}

Reject if $|U| > t_{0.95,7} = 1.895$. Since $1.979 > 1.895$, \textbf{reject $H_0$}.

\section*{9.6 \#1}

Two independent normal samples, equal variances. From Example 9.6.3:
Island Thorns ($n_X = 5$, $\bar{X} = 18.18$, $S_X^2 = 12.61$),
Ashley Rails ($n_Y = 5$, $\bar{Y} = 17.32$, $S_Y^2 = 1.043$).
Test $H_0\!: \mu_X = \mu_Y$ vs.\ $H_1\!: \mu_X \neq \mu_Y$ at $\alpha_0 = 0.05$.

Pooled variance and statistic:
\begin{align*}
    s_p^2 &= \tfrac{S_X^2 + S_Y^2}{n_X + n_Y - 2}
          = \tfrac{13.653}{8} \approx 1.707, \\
    s_p\sqrt{\tfrac{1}{n_X}+\tfrac{1}{n_Y}}
          &\approx 1.306\sqrt{0.4} \approx 0.826, \\
    U &= \tfrac{\bar{X}-\bar{Y}}{s_p\sqrt{1/n_X + 1/n_Y}}
       = \tfrac{0.86}{0.826} \approx 1.041 \;\sim\; t_8.
\end{align*}

Reject if $|U| > t_{0.975,8} = 2.306$. Since $1.041 < 2.306$, \textbf{fail to reject $H_0$}.

\section*{Q1}

$f(x\mid\theta) = 2(1-\theta)x + \theta$ on $[0,1]$, $0 \le \theta \le 2$.
Hypotheses (per 9.10 \#4): $H_0\!: \theta \ge 1$, $H_1\!: \theta < 1$.
Test rejects $H_0$ when $X > 1/3$.

Power function:
\begin{align*}
    \pi(\theta) &= \int_{1/3}^{1}\!\bigl[2(1-\theta)x + \theta\bigr]dx \\
                &= (1-\theta)\!\left(1 - \tfrac{1}{9}\right) + \theta\!\left(1 - \tfrac{1}{3}\right) \\
                &= \tfrac{8}{9} - \tfrac{2}{9}\theta.
\end{align*}

\textbf{Type I.} $\pi$ is decreasing, so on $H_0\!: \theta \ge 1$ the sup is at $\theta = 1$:
\[
    \alpha(\delta) = \pi(1) = \tfrac{8}{9} - \tfrac{2}{9} = \tfrac{2}{3}.
\]

\textbf{Type II.} $1 - \pi(\theta) = \tfrac{1}{9} + \tfrac{2}{9}\theta$ is increasing, so on $H_1\!: \theta < 1$ the sup is at $\theta \uparrow 1$:
\[
    \beta(\delta) = \tfrac{1}{9} + \tfrac{2}{9} = \tfrac{1}{3}.
\]
