\section*{Problem 8.7.11}

Write $\sigma_B^2 = \sigma^2$, so $\sigma_A^2 = 4\sigma^2$. By independence, $\operatorname{Var}(\bar{X}_m) = 4\sigma^2/m$ and $\operatorname{Var}(\bar{Y}_n) = \sigma^2/n$.

\textbf{(a)} For any $\alpha, m, n$,
\[
E[\hat{\theta}] = \alpha\,\theta + (1-\alpha)\,\theta = \theta,
\]
so $\hat{\theta}$ is unbiased \emph{for every} $\alpha \in \mathbb{R}$ and every $m,n \ge 1$ --- the unbiasedness does not constrain any of these.

\textbf{(b)}
\[
\operatorname{Var}(\hat{\theta}) = \frac{4\alpha^2 \sigma^2}{m} + \frac{(1-\alpha)^2\sigma^2}{n}.
\]
Differentiating w.r.t.\ $\alpha$ and setting to zero:
\[
\frac{8\alpha}{m} - \frac{2(1-\alpha)}{n} = 0
\;\Longrightarrow\;
8\alpha n = 2m(1-\alpha)
\;\Longrightarrow\;
\boxed{\alpha^* = \frac{m}{m + 4n}.}
\]

\section*{Problem 8.8.2}

Geometric with $f(x \mid p) = p(1-p)^x$, $x = 0,1,2,\dots$ and $E[X] = (1-p)/p$ (DeGroot \S5.5).
\[
\log f = \log p + x\log(1-p),
\qquad
\frac{\partial^2 \log f}{\partial p^2} = -\frac{1}{p^2} - \frac{x}{(1-p)^2}.
\]
Then
\[
I(p) = \frac{1}{p^2} + \frac{E[X]}{(1-p)^2}
= \frac{1}{p^2} + \frac{1}{p(1-p)}
= \boxed{\frac{1}{p^2(1-p)}.}
\]

\section*{Problem 8.8.7}

For Bernoulli$(p)$, $\log f = x\log p + (1-x)\log(1-p)$, so
\[
\frac{\partial^2 \log f}{\partial p^2} = -\frac{x}{p^2} - \frac{1-x}{(1-p)^2}
\;\Longrightarrow\;
I_1(p) = \frac{p}{p^2} + \frac{1-p}{(1-p)^2} = \frac{1}{p(1-p)}.
\]
For an i.i.d.\ sample of size $n$, $I_n(p) = n/[p(1-p)]$, so by the Cram\'er--Rao inequality any unbiased estimator $T$ of $p$ satisfies
\[
\operatorname{Var}(T) \ge \frac{1}{I_n(p)} = \frac{p(1-p)}{n}.
\]
Now $\bar{X}_n$ is unbiased ($E[\bar{X}_n] = p$) and
\[
\operatorname{Var}(\bar{X}_n) = \frac{\operatorname{Var}(X_1)}{n} = \frac{p(1-p)}{n},
\]
which equals the CRLB. So $\bar{X}_n$ attains the bound and is therefore efficient. \hfill $\blacksquare$

\section*{Problem 8.8.17}

For $X \sim \mathrm{Bin}(n,p)$ with $n$ known,
\[
\log f = \log\binom{n}{x} + x\log p + (n-x)\log(1-p),
\quad
\frac{\partial^2\log f}{\partial p^2} = -\frac{x}{p^2} - \frac{n-x}{(1-p)^2}.
\]
Note that $E[X] = np$ and $E[n-X] = n(1-p)$, so
\[
I(p) = \frac{np}{p^2} + \frac{n(1-p)}{(1-p)^2}
= \frac{n}{p} + \frac{n}{1-p}
= \frac{n[(1-p) + p]}{p(1-p)}
= \boxed{\frac{n}{p(1-p)}.}
\]

\section*{Problem 9.1.1(a)}

DeGroot's exponential is parameterized by rate $\beta$: $f(x \mid \beta) = \beta e^{-\beta x}$ for $x \ge 0$. The test rejects exactly when $X \ge 1$, so
\[
\pi(\beta) = P_\beta(X \ge 1) = \int_1^\infty \beta e^{-\beta x}\,dx = e^{-\beta}.
\]
\[
\boxed{\pi(\beta) = e^{-\beta} \quad \text{for all } \beta > 0.}
\]
